Child class overrides parent method with different variable return or parameter types 

/*
 * Protocol declared method aMethod with return type bool
 */
@protocol ExampleProtocol
- (BOOL) aMethodSelector;
@end


/*
 * Implementation declared a method with the same selector 'aMethodSelector:' but with return type VOID
 */
@implementation Example <ExampleProtocol>
- (VOID) aMethodSelector;
@end
Method in implementation is highlighted as it overwrites method 'aMethod:' with a different return type
